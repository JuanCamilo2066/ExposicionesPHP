      <?php
        class Persona {
          public $nombre = "";
          
          function __construct($nombre) {
              $this->nombre = $nombre;
          }
          
          public function bailar() {
            return "Estoy bailando";
          }
        }
        
        $yo = new Persona("Sabrina");
        if (is_a($yo, "Persona")) {
          echo "Soy una persona, ";
        }
        if (property_exists($yo, "nombre")) {
          echo "Tengo un nombre, ";
        }
        if (method_exists($yo, "bailar")) {
          echo "y se bailar.";
        }
      ?>
    </p>
